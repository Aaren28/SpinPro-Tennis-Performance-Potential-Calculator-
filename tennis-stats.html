function saveTennisStats() { const serve = document.getElementById("serve-speed").value; const firstServe = document.getElementById("first-serve").value; const errors = document.getElementById("unforced-errors").value; const win = document.getElementById("win-percentage").value; const years = document.getElementById("years-experience").value; const errorText = document.getElementById("tennisError"); if (!serve || !firstServe || !errors || !win || !years) { errorText.textContent = "⚠️ Please fill out all tennis stats before continuing."; return; } const data = { serve: +serve, firstServe: +firstServe, errors: +errors, win: +win, years: +years, }; localStorage.setItem("tennisStats", JSON.stringify(data)); window.location.href = "summary.html"; }
